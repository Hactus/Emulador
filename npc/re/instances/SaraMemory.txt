dali,139,118,4	script	Dimensional Device#sara	10007,{
	if (BaseLevel < 99) {
		mes "[Leon, O Aventureiro]";
		mes "Ei! Você é jovem demais para brincar com isso! Volte quando for ^0000FFLevel 99^000000!";
		close;
	}
	if ((saramemory < 1) || (saramemory == 2) || (saramemory == 3) || (saramemory == 5)) {
		mes "[Leon, O Aventureiro]";
		mes "Antes de brincar com isso; Porque não vir até aqui conversar comigo?";
		close;
	}

	// 20 hour cooldown
	set .@ins_sara_time,checkquest(15002,PLAYTIME);
	if (.@ins_sara_time == -1) {

		set .@party_id,getcharid(1);
		set .@p_name$,getpartyname(.@party_id);
		set .@md_name$,"Sara's Memories";

		if (!.@party_id) {
			mes "[Leon, O Aventureiro]";
			mes "Antes de entrar precisa-se organizar um grupo! Não sabe como? Digite /organize NomeDoGrupoAqui.";
			close;
		}

		if (getcharid(0) == getpartyleader(.@party_id,2))
			set .@menu$, "Ativar o mecânismo dimensional.:Utilizar mecânismo dimensional.:Cancelar.";
		else
			set .@menu$, ":Utilizar mecânismo dimensional.:Cancelar.";

		switch(select(.@menu$)) {
		case 1:
			if (instance_create(.@md_name$) < 0) {
				mes "Party Name: "+.@p_name$;
				mes "Party Leader: "+strcharinfo(0);
				mes "^0000ff"+.@md_name$+" ^000000- Falha ao Reservar!";
				close;
			}
			mes "^FF0000Mecânismo dimensional ativa-se sem problemas. Utilize-o para entrar na fenda no espaço-tempo.^000000";
			close;
		case 2:
			switch(instance_enter(.@md_name$)) {
			case IE_OTHER:
				mes "Um erro desconhecido ocorreu.";
				close;
			case IE_NOINSTANCE:
				mes "O calabouço Memorial"+.@md_name$+" Não existe.";
				mes "O líder do grupo não gerou a instância ainda.";
				close;
			case IE_NOMEMBER:
				mes "Apenas membro registrados podem adentrar a instância  "+.@md_name$+".";
				close;
			case IE_OK:
				mapannounce "dali",strcharinfo(0)+", membro do grupo "+.@p_name$+" Adentrou instância "+.@md_name$+".",bc_map,"0x00ff99";
				setquest 15002;
				end;
			}
		case 3:
			mes "^FF0000O líder do grupo não ativou o mecânismo.^000000";
			close;
		}
	} else if (.@ins_sara_time == 0 || .@ins_sara_time == 1) {
		mes "[Leon, O Aventureiro]";
		mes "Acredito que em aproximadamente 20 horas após entrar na fenda, a mesma estará estável o suficiente para que entre novamente.";
		close;
	}
	else {
		erasequest 15002;
		mes "^0000FF[ Memórias de Sarah ] Tempo de recarga completo. É possível utilizar o mecânismo dimensional novamente.^000000";
		close;
	}
}
